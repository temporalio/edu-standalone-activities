The fix — one line in activities.ts
Without (bug)
const headers: Record<string, string> = {
  'Content-Type': 'application/json',
  // no Idempotency-Key
};
With (fix)
const headers: Record<string, string> = {
  'Content-Type': 'application/json',
  'Idempotency-Key': `webhook:${req.eventId}`,
};
